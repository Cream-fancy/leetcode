## 13. 罗马数字转整数

```cpp
int romanToInt(string s) {
  map<char, int> mp;
  mp['I']=1, mp['V']=5, mp['X']=10, mp['L']=50;
  mp['C']=100, mp['D']=500, mp['M']=1000;
  int sz = s.size();
  int a[20] = {0};
  for(int i=0; i<sz; i++) a[i]=mp[s[i]];
  int cur, res = 0;
  for(int i=0; i<sz; i++){
    cur=a[i];
    if(cur>=a[i+1]) res+=cur;
    else res-=cur;
  }
  return res;
}
```

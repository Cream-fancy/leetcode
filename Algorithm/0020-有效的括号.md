## 20. 有效的括号

```cpp
bool isValid(string s) {
  stack<char> stk;
  map<char, char> mp;
  mp['(']=')',mp['[']=']',mp['{']='}';
  int sz = s.size();
  char ch;
  for(int i=0; i<sz; i++){
    ch = s[i];
    if(ch=='(' || ch=='[' || ch=='{'){
      stk.push(ch);
    }else{
      if(stk.size() && ch==mp[stk.top()]){
        stk.pop();
      }else{
        return false;
      }
    }
  }
  return !stk.size();
}
```

## 1190. 反转每对括号间的子串

```cpp
string reverseParentheses(string s) {
  stack<char> res;
  queue<char> q;
  int sz = s.size();
  char ch, top;
  for(int i=sz-1; i>=0; i--){
    if(s[i]=='('){
      top=res.top();
      while(top!=')'){
        res.pop();
        q.push(top);
        top=res.top();
      }
      res.pop();
      while(q.size()){
        res.push(q.front()), q.pop();
      }
    }else{
      res.push(s[i]);
    }
  }
  string ss = "";
  while(res.size()){
    ss += res.top();
    res.pop();
  }
  return ss;
}
```
